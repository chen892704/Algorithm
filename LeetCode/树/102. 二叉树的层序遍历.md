# 102. 二叉树的层序遍历

标签（空格分隔）： 树 队列

---

给你一个二叉树，请你返回其按 层序遍历 得到的节点值。 （即逐层地，从左到右访问所有节点）。


## 解题思路
用队列 BFS

## 代码实现
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) 
    {
        vector<vector<int> > ans;
        if (root == nullptr)
            return ans;
        
        queue<TreeNode*> que;
        que.push(root);
        while (!que.empty())
        {
            int nSize = que.size();
            vector<int> tmp;
            tmp.reserve(nSize);
            for (int i=0; i<nSize; ++i)
            {
                TreeNode *p = que.front();
                que.pop();
                if (p->left)
                    que.push(p->left);
                if (p->right)
                    que.push(p->right);
                tmp.push_back(p->val);
            }
            ans.push_back(tmp);
        }
        return ans;
    }
};
```
